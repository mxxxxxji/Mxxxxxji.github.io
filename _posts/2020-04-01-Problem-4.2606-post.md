#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <vector>
#define MAX 101
using namespace std;

int n, m;
vector<int> a[MAX];
bool c[MAX];
int cnt;
void dfs(int start) {
	if (c[start]) return;
	c[start] = true;
	cnt++;
	for (int i = 0; i < a[start].size(); i++) {
		int y = a[start][i];
		dfs(y);
	}
}

int main(void) {
	scanf("%d %d", &n, &m);
	for (int i = 0; i < m; i++) {
		int b, c;
		scanf("%d %d", &b, &c);
		a[b].push_back(c);
		a[c].push_back(b);
	}
	dfs(1);
	printf("%d\n", cnt-1);
	return 0;
}
