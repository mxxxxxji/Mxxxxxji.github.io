#include <iostream>
#include <string>
using namespace std;

int main()
{
	int num;
	string value = "*";
	cin >> num;
	for (int i = 0; i < num*2; i++) { 
		if (i % 2 == 0) {
			value = "*";
		}else {
			value = " ";
		}
		for (int i = 0; i < num; i++) {
			cout << value;
			if (value == "*") {
				value = " ";
			}else {
				value = "*";
			}
		}
		cout << "\n";
	}
}
